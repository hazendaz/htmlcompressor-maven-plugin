58       * @throws MojoExecutionException
59       *             the mojo execution exception
60       */
61      @Test
62      void testExecute() throws MojoExecutionException {
63          LOG.info("Testing mojo execution...");
64  
65          HtmlCompressorMojo htmlCompressorMojo = new HtmlCompressorMojo();
66          htmlCompressorMojo.setSrcFolder("src/test/resources/html");
67          htmlCompressorMojo.setJavascriptHtmlSpriteIntegrationFile("src/test/resources/html/integration.js");
68          htmlCompressorMojo.setTargetFolder("target/htmlcompressor/html");
69          htmlCompressorMojo.execute();
70  
71          // Verify output files were produced
72          Assertions.assertTrue(Files.exists(Path.of("target/htmlcompressor/html/templates/Template1.html")),
73                  "Compressed Template1.html should exist in target folder");
74          Assertions.assertTrue(Files.exists(Path.of("target/htmlcompressor/html/templates/Template2.html")),
75                  "Compressed Template2.html should exist in target folder");
76  
77          LOG.info("Passed");
78      }
